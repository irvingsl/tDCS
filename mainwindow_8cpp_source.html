172 bool MainWindow::writeToSerial(QString msg)
173 {
174  const qint64 bytesWritten = serialPort.write(msg.toUtf8());
175  if (bytesWritten == -1)
176  {
177  qDebug() << QObject::tr("Failed to write the data to port %1, error: %2")
178  .arg(serialPort.portName()).arg(serialPort.errorString()) << endl;
179  return false;
180  }
181  else if (bytesWritten != msg.size())
182  {
183  qDebug() << QObject::tr("Failed to write all the data to port %1, error: %2, sizes: written: %3, in msg: %4")
184  .arg(serialPort.portName()).arg(serialPort.errorString()).arg(bytesWritten).arg(msg.size()) << endl;
185  }
186  else
187  {
188  qDebug() << tr("Message of size %1 succesfully sent.").arg(bytesWritten);
189  return true;
190  }
191  return false; //To avoid warning (non-void function)
192 }
